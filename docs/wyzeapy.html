 60    async def login(
 61        self, email, password, key_id, api_key, token: Optional[Token] = None
 62    ):
 63        """
 64        Logs the user in and retrieves the users token
 65
 66        :param email: Users email
 67        :param password: Users password
 68        :param key_id: Key ID for third-party API access
 69        :param api_key: API Key for third-party API access
 70        :param token: Users existing token from a previous session
 71
 72        :raises:
 73            TwoFactorAuthenticationEnabled: indicates that the account has 2fa enabled
 74        """
 75
 76        self._email = email
 77        self._password = password
 78        self._key_id = key_id
 79        self._api_key = api_key
 80
 81        try:
 82            self._auth_lib = await WyzeAuthLib.create(
 83                email, password, key_id, api_key, token, self.execute_token_callbacks
 84            )
 85            if token:
 86                # User token supplied, refresh on startup
 87                await self._auth_lib.refresh()
 88            else:
 89                await self._auth_lib.get_token_with_username_password(
 90                    email, password, key_id, api_key
 91                )
 92            self._service = BaseService(self._auth_lib)
 93        except TwoFactorAuthenticationEnabled as error:
 94            raise error

110    async def execute_token_callbacks(self, token: Token):
111        """
112        Sends the token to the registered callback functions.
113
114        :param token: Users token object
115
116        """
117        for callback in self._token_callbacks:
118            if iscoroutinefunction(callback):
119                await callback(token)
120            else:
121                callback(token)

124        """

127        :param callback_function: A callback function which expects a token object
128
129        """
130        self._token_callbacks.append(callback_function)

138        """
139        self._token_callbacks.remove(callback_function)
